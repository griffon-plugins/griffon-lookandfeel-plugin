059     private static void addSyntheticaLookAndFeelDescriptor(@Nonnull String displayName, @Nonnull String lookAndFeelClassName) {
060         try {
061             SUPPORTED_LAFS.add(new SyntheticaLookAndFeelDescriptor(displayName, lookAndFeelClassName));
062         } catch (IllegalArgumentException e) {
063             // ignore
064         }
065     }
066 
067     public SyntheticaLookAndFeelHandler() {
068         System.out.println(SUPPORTED_LAFS);
069         this.supportedDescriptors = SUPPORTED_LAFS.toArray(new SyntheticaLookAndFeelDescriptor[SUPPORTED_LAFS.size()]);
070     }
071 
072     @Override
073     public boolean handles(@Nonnull LookAndFeel laf) {
074         requireNonNull(laf, "Argument 'laf' must not be null");
075         for (LookAndFeelDescriptor descriptor : supportedDescriptors) {
076             if (descriptor.getLookAndFeel().getClass().getName().equals(laf.getClass().getName())) {
077                 return true;
078             }
079         }
080         return false;
081     }
082 
083     @Override
084     public boolean handles(@Nonnull LookAndFeelDescriptor descriptor) {
085         requireNonNull(descriptor, "Argument 'descriptor' must not be null");
086         return descriptor instanceof SyntheticaLookAndFeelDescriptor;
087     }
088 
089     @Nonnull
090     @Override
091     public LookAndFeelDescriptor[] getSupportedLookAndFeelDescriptors() {
092         return supportedDescriptors;
093     }
094 
095     private static class SyntheticaLookAndFeelDescriptor extends DefaultLookAndFeelDescriptor {
096         public SyntheticaLookAndFeelDescriptor(@Nonnull String displayName, @Nonnull String lookAndFeelClassName) {
097             super("synthetica-" + displayName.toLowerCase(), displayName, lookAndFeelClassName);
098         }
099     }
