076     private static Skin loadSkin(String name) {
077         try {
078             URL url = SkinLookAndFeelHandler.class.getResource("/skinlnf/themepacks/" + name + ".themepack");
079             return SkinLookAndFeel.loadThemePack(url);
080         } catch (Exception e) {
081             throw new IllegalStateException("Cannot load skin theme " + name);
082         }
083     }
084 
085     private final LookAndFeelDescriptor[] supportedDescriptors;
086 
087     public SkinLookAndFeelHandler() {
088         this.supportedDescriptors = SUPPORTED_LAFS.toArray(new SkinLookAndFeelDescriptor[SUPPORTED_LAFS.size()]);
089     }
090 
091     @Override
092     public boolean handles(@Nonnull LookAndFeel laf) {
093         requireNonNull(laf, "Argument 'laf' must not be null");
094         for (LookAndFeelDescriptor descriptor : supportedDescriptors) {
095             if (descriptor.getLookAndFeel().getClass().getName().equals(laf.getClass().getName())) {
096                 return true;
097             }
098         }
099         return false;
100     }
101 
102     @Override
103     public boolean handles(@Nonnull LookAndFeelDescriptor descriptor) {
104         requireNonNull(descriptor, "Argument 'descriptor' must not be null");
105         return descriptor instanceof SkinLookAndFeelDescriptor;
106     }
107 
108     @Nonnull
109     @Override
110     public LookAndFeelDescriptor[] getSupportedLookAndFeelDescriptors() {
111         return supportedDescriptors;
112     }
113 
114     private static class SkinLookAndFeelDescriptor extends DefaultLookAndFeelDescriptor {
115         private final Skin skin;
116 
117         public SkinLookAndFeelDescriptor(@Nonnull String displayName, @Nonnull Skin skin) {
118             super(createIdentifier(displayName), displayName, new SkinLookAndFeel());
119             this.skin = skin;
120             SkinLookAndFeel.setSkin(skin);
121         }
122 
123         @Override
124         public void install() {
125             SkinLookAndFeel.setSkin(skin);
126             super.install();
127         }
128 
129         @Override
130         public boolean isCurrent() {
131             return super.isCurrent() && SkinLookAndFeel.getSkin().equals(skin);
132         }
133 
134         private static String createIdentifier(String displayName) {
135             return "skin-" + displayName.toLowerCase();
136         }
137     }
